    // простая защита от XSS

            return m;
        }).replace(/[\uD800-\uDBFF][\uDC00-\uDFFF]/g, function(c) {
            return c;

        modal.style.display = "flex";
    }

    function closeModal() {
        modal.style.display = "none";
        pendingDownloadGame = null;
    }

    function confirmDownload() {
        if (pendingDownloadGame) {
            // Используем уникальный идентификатор apkFileRef из данных
            const ref = pendingDownloadGame.apkFileRef || `${pendingDownloadGame.title.toLowerCase()}.apk`;
            downloadApkFile(pendingDownloadGame.title, ref);
            closeModal();
            // Показываем дополнительный тост (просто в консоль, но можно сделать всплывашку)
            const notif = document.createElement('div');
            notif.innerText = `✔ Начинается загрузка ${pendingDownloadGame.title}.apk`;
            notif.style.position = 'fixed';
            notif.style.bottom = '20px';
            notif.style.left = '20px';
            notif.style.backgroundColor = '#10b981';
            notif.style.color = 'white';
            notif.style.padding = '12px 20px';
            notif.style.borderRadius = '40px';
            notif.style.zIndex = '1100';
            notif.style.fontWeight = '500';
            notif.style.boxShadow = '0 6px 14px black';
            document.body.appendChild(notif);
            setTimeout(() => notif.remove(), 2500);
        } else {
            closeModal();
        }
    }

    // Генерация кнопок категорий
    function renderCategoryFilters() {
        categoryFiltersDiv.innerHTML = allCategories.map(cat => `
            <button class="cat-btn ${currentCategory === cat ? 'active' : ''}" data-cat="${cat}">${cat}</button>
        `).join('');
        document.querySelectorAll('.cat-btn').forEach(btn => {
            btn.addEventListener('click', (e) => {
                currentCategory = btn.getAttribute('data-cat');
                renderCategoryFilters();   // перерисовать активный класс
                renderGames();
            });
        });
    }

    // События поиска
    searchInput.addEventListener('input', (e) => {
        searchQuery = e.target.value;
        renderGames();
    });

    // Закрытие модалки по крестику/отмене
    document.getElementById("modalCancelBtn").addEventListener("click", closeModal);
    document.getElementById("modalConfirmBtn").addEventListener("click", confirmDownload);
    window.addEventListener("click", (e) => {
        if (e.target === modal) closeModal();
    });

    // инициализация
    function init() {
        renderCategoryFilters();
        renderGames();
    }
    init();
